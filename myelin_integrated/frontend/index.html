API Access

Integrate Myelin directly into your AI pipeline using our API. No UI complexity — just clean access points. 

import requests

# Audit a conversation
response = requests.post(
    "http://localhost:8000/api/v1/audit/conversation",
    json={
        "user_input": "Your user message",
        "bot_response": "AI response to check",
        "source_text": "Optional reference text"
    }
)

result = response.json()
print(f"Decision: {result['overall']['decision']}")
print(f"Risk Level: {result['overall']['risk_level']}")
